import cv2

def preprocess_image(image_path, target_size=(224, 224)):
  """
  Loads, resizes, and normalizes an image.

  Args:
      image_path:r'C:\beproject\Potato'
      target_size: (224,224)

  Returns:
      Resized and normalized image as a NumPy array.
  """
  img = cv2.imread(image_path)
  img = cv2.resize(img, target_size)  # Resize
  img = img.astype('float32') / 255.0  # Normalize
  return img
