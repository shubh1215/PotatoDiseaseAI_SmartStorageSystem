from tensorflow.keras.applications import VGG16
from tensorflow.keras.layers import GlobalAveragePooling2D, Dense

# Define input shape (adjust based on your data)
input_shape = (224, 224, 3)

# Load pre-trained VGG16, excluding the top classification layers
vgg16 = VGG16(weights='imagenet', include_top=False, input_shape=input_shape)

# Freeze pre-trained VGG16 layers
for layer in vgg16.layers:
    layer.trainable = False

# Add custom head
x = vgg16.output
x = GlobalAveragePooling2D()(x)
x = Dense(1024, activation='relu')(x)  # Example dense layer
predictions = Dense(num_classes, activation='softmax')(x)

# Create the final model
model = tf.keras.Model(inputs=vgg16.input, outputs=predictions)
